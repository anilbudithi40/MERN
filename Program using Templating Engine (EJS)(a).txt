const express = require("express");
const app = express();

// set EJS
app.set("view engine", "ejs");

// route
app.get("/", (req, res) => {
    res.render("home", {
        title: "Home Page",
        message: "Welcome to Express with EJS"
    });
});

app.listen(3000, () => {
    console.log("Server running on port 3000");
});
