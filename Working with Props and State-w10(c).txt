import React from "react";
import ReactDOM from "react-dom/client";

// child component
function Greeting(props) {
  return <h2>Hello, {props.name}</h2>;
}

// parent component
function App() {
  return (
    <div>
      <Greeting name="Alice" />
      <Greeting name="Bob" />
    </div>
  );
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
