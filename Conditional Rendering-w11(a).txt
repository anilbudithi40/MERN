import React from "react";
import ReactDOM from "react-dom/client";

function App() {
  const isLoggedIn = true;

  return (
    <div>
      <h1>Conditional Rendering</h1>
      {isLoggedIn ? (
        <p>Welcome back, User!</p>
      ) : (
        <p>Please log in</p>
      )}
    </div>
  );
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
